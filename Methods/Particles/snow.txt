
namespace SnowAnimation
{
    public partial class Form1 : Form
    {
        private List<Snowflake> snowflakes = new List<Snowflake>();
        private Timer timer = new Timer(); //https://discord.gg/wQjFQ57Jfn  

        public Form1()
        {
            InitializeComponent();
            this.DoubleBuffered = true; // reduce flicker
            this.BackColor = Color.Black; // winter night background

            // Create initial snowflakes //https://discord.gg/wQjFQ57Jfn  
            for (int i = 0; i < 150; i++)
            {
                snowflakes.Add(new Snowflake(this.ClientSize.Width));
            }

            // Setup timer
            timer.Interval = 30; // ~33 FPS
            timer.Tick += Timer_Tick;
            timer.Start(); //https://discord.gg/wQjFQ57Jfn  
        }

        private void Timer_Tick(object sender, EventArgs e)
        {
            foreach (var flake in snowflakes)
            {
                flake.Move(this.ClientSize.Height, this.ClientSize.Width);
            }
            this.Invalidate(); // redraw
        } //https://discord.gg/wQjFQ57Jfn  

        protected override void OnPaint(PaintEventArgs e)
        {
            base.OnPaint(e);
            foreach (var flake in snowflakes)
            {
                e.Graphics.FillEllipse(Brushes.White, flake.X, flake.Y, flake.Size, flake.Size);
            }
        }
    } //https://discord.gg/wQjFQ57Jfn  

    public class Snowflake
    {
        public int X { get; set; }
        public int Y { get; set; }
        public int Size { get; set; }
        public int Speed { get; set; } //https://discord.gg/wQjFQ57Jfn  

        private static Random rand = new Random();

        public Snowflake(int formWidth)
        {
            X = rand.Next(0, formWidth);
            Y = rand.Next(-50, 0); // start above form
            Size = rand.Next(2, 8); // random size
            Speed = rand.Next(2, 6); // random fall speed //https://discord.gg/wQjFQ57Jfn  
        }

        public void Move(int formHeight, int formWidth)
        {
            Y += Speed;
            X += rand.Next(-1, 2); // slight wind drift

            if (Y > formHeight)
            {
                // Reset snowflake to top //https://discord.gg/wQjFQ57Jfn  
                Y = rand.Next(-50, 0);
                X = rand.Next(0, formWidth); //https://discord.gg/wQjFQ57Jfn  
            }
        }
    }
}
