   public partial class Form1 : Form
    { //https://discord.gg/wQjFQ57Jfn  
        private const int ParticleCount = 150;
        private readonly Random _random = new Random();
        private readonly PointF[] _particlePositions = new PointF[ParticleCount];
        private readonly PointF[] _particleTargetPositions = new PointF[ParticleCount];
        private readonly float[] _particleSpeeds = new float[ParticleCount];
        private readonly float[] _particleSizes = new float[ParticleCount];
        private readonly float[] _particleRotations = new float[ParticleCount];
        private Color particleColor = Color.Red; // default
 //https://discord.gg/wQjFQ57Jfn  


        public Form1()
        {
            InitializeComponent();

            DoubleBuffered = true;
            InitializeParticles();

            Timer timer = new Timer { Interval = 10 };
            timer.Tick += (sender, args) => { UpdateParticles(); Invalidate(); };
            timer.Start();

        }
        private System.Windows.Forms.Button colorButton; //https://discord.gg/wQjFQ57Jfn  

        private void InitializeParticles()
        {
            Size screenSize = Screen.PrimaryScreen.Bounds.Size;
            for (int i = 0; i < ParticleCount; i++)
            {
                _particlePositions[i] = new PointF(_random.Next(screenSize.Width), _random.Next(screenSize.Height));
                _particleTargetPositions[i] = new PointF(_random.Next(screenSize.Width), screenSize.Height + 50);
                _particleSpeeds[i] = 2 + _random.Next(15);
                _particleSizes[i] = _random.Next(4, 8);
                _particleRotations[i] = 0; //https://discord.gg/wQjFQ57Jfn  
            }
        }
        private void colorButton_Click(object sender, EventArgs e)
        {
            using (ColorDialog colorDialog = new ColorDialog())
            {
                if (colorDialog.ShowDialog() == DialogResult.OK)
                {
                    particleColor = colorDialog.Color;
                }
            }
        }

        private void UpdateParticles()
        { //https://discord.gg/wQjFQ57Jfn  
            Size screenSize = Screen.PrimaryScreen.Bounds.Size;
            for (int i = 0; i < ParticleCount; i++)
            {
                float deltaTime = 1f / 60;
                _particlePositions[i] = Lerp(_particlePositions[i], _particleTargetPositions[i], deltaTime * (_particleSpeeds[i] / 60));
                _particleRotations[i] += 0.02f;

                if (_particlePositions[i].Y > screenSize.Height)
                {
                    _particlePositions[i] = new PointF(_random.Next(screenSize.Width), -10);
                    _particleRotations[i] = 0;
                } //https://discord.gg/wQjFQ57Jfn  
            }
        }
        private PointF Lerp(PointF start, PointF end, float t)
        {
            return new PointF(start.X + (end.X - start.X) * t, start.Y + (end.Y - start.Y) * t);
        }
        protected override void OnPaint(PaintEventArgs e)
        {
            base.OnPaint(e);
            e.Graphics.SmoothingMode = System.Drawing.Drawing2D.SmoothingMode.AntiAlias;

            for (int i = 0; i < ParticleCount; i++) //https://discord.gg/wQjFQ57Jfn  
            {
                DrawTriangleWithGlow(e.Graphics, _particlePositions[i], _particleSizes[i], _particleRotations[i]);
            }
        }
        private void DrawTriangleWithGlow(Graphics g, PointF center, float size, float rotation)
        {
            PointF[] triangle = new PointF[3];
            for (int i = 0; i < 3; i++)
            {
                float angle = rotation + i * (float)(2 * Math.PI / 3);
                triangle[i] = new PointF(
                    center.X + size * (float)Math.Cos(angle),
                    center.Y + size * (float)Math.Sin(angle)
                );
            } //https://discord.gg/wQjFQ57Jfn  

            // Glow layers
            for (int j = 0; j < 4; j++)
            {
                int alpha = 40 - j * 8;
                float glowSize = size + j * 4;
                using (Brush glow = new SolidBrush(Color.FromArgb(alpha, particleColor)))

                {
                    g.FillEllipse(glow, center.X - glowSize / 2, center.Y - glowSize / 2, glowSize, glowSize);
                }
            } //https://discord.gg/wQjFQ57Jfn  

            // Red triangle
            using (SolidBrush brush = new SolidBrush(particleColor))

            {
                g.FillPolygon(brush, triangle); //https://discord.gg/wQjFQ57Jfn  
            }
        }
